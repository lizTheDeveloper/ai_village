/**
 * Simple check: verify governance buildings are in the blueprint registry source code
 */

import { readFileSync } from 'fs';

console.log('=== Governance Buildings Source Code Check ===\n');

const registrySource = readFileSync('./packages/core/src/buildings/BuildingBlueprintRegistry.ts', 'utf-8');

const governanceBuildings = [
  'town_hall',
  'census_bureau',
  'granary',
  'weather_station',
  'health_clinic',
  'meeting_hall',
  'watchtower',
  'labor_guild',
  'archive',
];

let allFound = true;

governanceBuildings.forEach(id => {
  // Check if building ID exists in the source
  const pattern = new RegExp(`id:\\s*['"]${id}['"]`, 'i');
  const found = pattern.test(registrySource);

  if (found) {
    console.log(`‚úì ${id} - Found in source code`);
  } else {
    console.error(`‚úó ${id} - NOT FOUND in source code`);
    allFound = false;
  }
});

// Check that registerGovernanceBuildings() is called
const calledInDefaults = registrySource.includes('this.registerGovernanceBuildings()');

console.log('\n=== Integration Check ===\n');
if (calledInDefaults) {
  console.log('‚úì registerGovernanceBuildings() is called in registerDefaults()');
} else {
  console.error('‚úó registerGovernanceBuildings() is NOT called in registerDefaults()');
  allFound = false;
}

// Check building categories
console.log('\n=== Expected Categories ===\n');

const categoryChecks = [
  { id: 'town_hall', expectedCategory: 'community' },
  { id: 'census_bureau', expectedCategory: 'community' },
  { id: 'granary', expectedCategory: 'storage' },
  { id: 'weather_station', expectedCategory: 'community' },
  { id: 'health_clinic', expectedCategory: 'community' },
  { id: 'meeting_hall', expectedCategory: 'community' },
  { id: 'watchtower', expectedCategory: 'community' },
  { id: 'labor_guild', expectedCategory: 'community' },
  { id: 'archive', expectedCategory: 'research' },
];

categoryChecks.forEach(({ id, expectedCategory }) => {
  // Find the building registration block
  const idPattern = new RegExp(`id:\\s*['"]${id}['"][\\s\\S]*?category:\\s*['"]([^'"]+)['"]`);
  const match = registrySource.match(idPattern);

  if (match && match[1] === expectedCategory) {
    console.log(`‚úì ${id} ‚Üí category: '${match[1]}'`);
  } else if (match) {
    console.warn(`‚ö†  ${id} ‚Üí category: '${match[1]}' (expected: '${expectedCategory}')`);
  } else {
    console.error(`‚úó ${id} ‚Üí category not found`);
  }
});

console.log('\n=== Summary ===\n');
if (allFound) {
  console.log('‚úÖ SUCCESS: All 9 governance buildings are defined in the source code!');
  console.log('\nüìç Buildings are located in the following categories:');
  console.log('   ‚Ä¢ COMMUNITY: Town Hall, Census Bureau, Weather Station, Health Clinic,');
  console.log('                Meeting Hall, Watchtower, Labor Guild (7 buildings)');
  console.log('   ‚Ä¢ STORAGE: Granary (1 building)');
  console.log('   ‚Ä¢ RESEARCH: Archive (1 building)');
  console.log('\nüéÆ In-game access:');
  console.log('   1. Press "B" to open building menu');
  console.log('   2. Click category tabs at top of menu');
  console.log('   3. Community tab contains most governance buildings');
  process.exit(0);
} else {
  console.log('‚ùå FAILURE: Some governance buildings are missing or misconfigured!');
  process.exit(1);
}
