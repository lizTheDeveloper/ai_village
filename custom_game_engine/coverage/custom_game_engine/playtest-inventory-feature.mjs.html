import { chromium } from '@playwright/test';

const screenshotDir = '/Users/annhoward/src/ai_village/agents/autonomous-dev/work-orders/agent-inventory-display/screenshots';

async function main() {
  console.log('Starting browser for inventory playtest...');
  const browser = await chromium.launch({
    headless: false,
    slowMo: 500 // Slow down actions for visibility
  });
  const page = await browser.newPage();

  try {
    console.log('\n=== CRITERION 1: Inventory Section Appears ===');

    // Navigate to game
    console.log('Navigating to game...');
    await page.goto('http://localhost:3000', { waitUntil: 'networkidle' });

    // Wait for game to load
    console.log('Waiting for game to load...');
    await page.waitForTimeout(3000);

    // Take initial screenshot
    await page.screenshot({ path: `${screenshotDir}/criterion-1-initial.png`, fullPage: true });
    console.log('âœ“ Initial screenshot taken');

    // Get canvas element
    const canvas = await page.locator('canvas').first();

    // Try clicking at various positions to find an agent
    console.log('Attempting to click on agents...');
    const clickPositions = [
      { x: 300, y: 300, name: 'top-left' },
      { x: 500, y: 400, name: 'center' },
      { x: 700, y: 500, name: 'bottom-right' },
      { x: 600, y: 300, name: 'top-center' },
      { x: 400, y: 500, name: 'bottom-left' }
    ];

    for (const pos of clickPositions) {
      console.log(`  Clicking at ${pos.name} (${pos.x}, ${pos.y})...`);
      await canvas.click({ position: { x: pos.x, y: pos.y } });
      await page.waitForTimeout(1000);
      await page.screenshot({
        path: `${screenshotDir}/criterion-1-click-${pos.name}.png`,
        fullPage: true
      });
    }

    // Take final screenshot
    await page.screenshot({ path: `${screenshotDir}/criterion-1-final.png`, fullPage: true });

    console.log('\n=== Analysis Required ===');
    console.log('Check screenshots to see if AgentInfoPanel appeared on the right side.');
    console.log('Expected: Panel with sections for Agent Info, Needs, and INVENTORY');
    console.log('\nScreenshots saved to:', screenshotDir);

    console.log('\n=== Keeping browser open for manual inspection ===');
    console.log('You can manually click on agents to verify the behavior.');
    console.log('Press Ctrl+C when done.');

    // Keep browser open for manual testing
    await page.waitForTimeout(300000); // 5 minutes

  } catch (error) {
    console.error('Error during playtest:', error);
    await page.screenshot({ path: `${screenshotDir}/error-state.png`, fullPage: true });
  } finally {
    await browser.close();
  }
}

main().catch(console.error);
