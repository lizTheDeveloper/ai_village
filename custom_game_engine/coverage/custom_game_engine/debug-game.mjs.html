import { chromium } from 'playwright';

async function debug() {
  const browser = await chromium.launch({ headless: false });
  const page = await browser.newPage();

  page.on('console', msg => {
    const text = msg.text();
    if (text.includes('[Demo]') || text.includes('ERROR') || text.includes('ceremony') || text.includes('soul') || text.includes('hidden')) {
      console.log('BROWSER:', text);
    }
  });
  page.on('pageerror', err => console.log('PAGE ERROR:', err.message));
  page.on('dialog', async dialog => {
    console.log('ALERT:', dialog.message());
    await dialog.accept();
  });

  await page.goto('http://localhost:3000/');
  console.log('Page loaded');
  await page.waitForTimeout(2000);

  // Step 1: Select preset and click Start Game
  console.log('\n=== Step 1: Settings ===');
  await page.evaluate(() => {
    const selects = document.querySelectorAll('select');
    for (const select of selects) {
      for (const opt of select.options) {
        if (opt.textContent?.includes('Cooperative')) {
          select.value = opt.value;
          select.dispatchEvent(new Event('change', { bubbles: true }));
          return;
        }
      }
    }
  });
  await page.waitForTimeout(500);

  const startBtn = await page.$('#settings-save-btn');
  if (startBtn) {
    await startBtn.click();
    console.log('Clicked Start Game');
  }
  await page.waitForTimeout(2000);

  // Step 2: Navigate through magic and story steps
  console.log('\n=== Step 2: Universe Config ===');
  for (let step = 0; step < 3; step++) {
    const btns = await page.$$eval('button', btns => btns.map(b => b.textContent?.trim()));
    console.log('Buttons: ' + btns.slice(0, 5).join(', '));

    const nextClicked = await page.evaluate(() => {
      const buttons = Array.from(document.querySelectorAll('button'));
      for (const btn of buttons) {
        const text = btn.textContent || '';
        if (text.includes('Next')) {
          btn.click();
          return text.trim();
        }
      }
      return null;
    });

    if (nextClicked) {
      console.log('Clicked: ' + nextClicked);
      await page.waitForTimeout(2000);
    } else {
      break;
    }
  }

  // Step 3: Soul ceremonies - need to click Continue for each of 5 souls
  console.log('\n=== Step 3: Soul Ceremonies ===');
  let ceremonyCount = 0;
  const maxCeremonies = 10;

  for (let i = 0; i < 60; i++) {  // Up to 2 minutes
    await page.waitForTimeout(2000);

    // Check current state
    const state = await page.evaluate(() => {
      const buttons = Array.from(document.querySelectorAll('button')).map(b => b.textContent?.trim());
      const hasUniverseConfig = !!document.getElementById('universe-config-screen');
      const configVisible = hasUniverseConfig ?
        document.getElementById('universe-config-screen')?.style.display !== 'none' : false;
      return { buttons, configVisible };
    });

    // Check if universe config is hidden (success!)
    if (!state.configVisible && ceremonyCount > 0) {
      console.log('SUCCESS! Universe config screen is now hidden!');
      break;
    }

    // Try to click Continue
    const clicked = await page.evaluate(() => {
      const btn = Array.from(document.querySelectorAll('button'))
        .find(b => b.textContent?.trim() === 'Continue');
      if (btn) {
        btn.click();
        return true;
      }
      return false;
    });

    if (clicked) {
      ceremonyCount++;
      console.log('Ceremony ' + ceremonyCount + ': clicked Continue');
    }

    // Log progress every 5 iterations
    if (i > 0 && i % 5 === 0) {
      console.log('Iteration ' + i + ', ceremonies: ' + ceremonyCount + ', configVisible: ' + state.configVisible);
    }
  }

  console.log('Total ceremonies completed: ' + ceremonyCount);

  // Final state check
  await page.waitForTimeout(3000);
  await page.screenshot({ path: '/tmp/game-final.png' });

  const finalState = await page.evaluate(() => {
    const configScreen = document.getElementById('universe-config-screen');
    return {
      url: location.href,
      configScreenExists: !!configScreen,
      configScreenVisible: configScreen ? configScreen.style.display !== 'none' : false,
      canvasVisible: document.querySelector('canvas')?.offsetParent !== null,
      bodyText: document.body.innerText.substring(0, 300)
    };
  });

  console.log('\n=== Final State ===');
  console.log(JSON.stringify(finalState, null, 2));

  if (!finalState.configScreenVisible && finalState.canvasVisible) {
    console.log('\n*** SUCCESS: Game is visible and config screen is hidden! ***');
  } else {
    console.log('\n*** ISSUE: Config screen still visible or canvas not showing ***');
  }

  await browser.close();
}

debug().catch(console.error);
