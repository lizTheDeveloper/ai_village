/**
 * Find and Fix Corrupted Souls - Browser Console Script
 *
 * Run this in the browser console after the game loads:
 * 1. Open DevTools (F12)
 * 2. Paste this entire script
 * 3. Run: findCorruptedSouls()
 * 4. For each corrupted soul, run: fixSoul('entity-id')
 */

function findCorruptedSouls() {
  if (typeof game === 'undefined' || !game.world) {
    console.error('‚ùå Game not loaded yet. Wait for game to start.');
    return;
  }

  console.log('üîç Searching for souls with <think> tags...\n');

  const corrupted = [];

  // Find all souls with creation events
  const souls = game.world.query()
    .with('soul_creation_event')
    .executeEntities();

  console.log(`Found ${souls.length} total souls\n`);

  for (const soul of souls) {
    const event = soul.getComponent('soul_creation_event');
    const identity = soul.getComponent('soul_identity');
    const trueName = identity?.true_name || 'Unknown';

    // Check creationDebate statements for <think> tags
    if (event.creationDebate?.statements) {
      for (let i = 0; i < event.creationDebate.statements.length; i++) {
        const stmt = event.creationDebate.statements[i];
        if (stmt.statement && stmt.statement.includes('<think>')) {
          corrupted.push({
            id: soul.id,
            trueName,
            statementIndex: i,
            fate: stmt.fate,
            preview: stmt.statement.substring(0, 150)
          });
        }
      }
    }

    // Check conversationTranscript for thoughts field
    if (event.conversationTranscript) {
      for (let i = 0; i < event.conversationTranscript.length; i++) {
        const exchange = event.conversationTranscript[i];
        if (exchange.thoughts && exchange.thoughts.trim().length > 0) {
          corrupted.push({
            id: soul.id,
            trueName,
            transcriptIndex: i,
            speaker: exchange.speaker,
            thoughtsPreview: exchange.thoughts.substring(0, 150)
          });
        }
      }
    }
  }

  if (corrupted.length === 0) {
    console.log('‚úÖ No corrupted souls found!\n');
    return [];
  }

  console.log(`‚ö†Ô∏è  Found ${corrupted.length} corruption points:\n`);

  // Group by soul ID
  const bySoul = {};
  for (const c of corrupted) {
    if (!bySoul[c.id]) {
      bySoul[c.id] = { id: c.id, trueName: c.trueName, issues: [] };
    }
    bySoul[c.id].issues.push(c);
  }

  const soulIds = Object.keys(bySoul);
  console.log(`Affects ${soulIds.length} souls:\n`);

  for (const soulId of soulIds) {
    const soul = bySoul[soulId];
    console.log(`\nü™¶ Soul: "${soul.trueName}" (${soul.id.substring(0, 8)}...)`);
    console.log(`   Issues: ${soul.issues.length}`);

    for (const issue of soul.issues) {
      if (issue.statementIndex !== undefined) {
        console.log(`   - Statement[${issue.statementIndex}] (${issue.fate}): Has <think> tag`);
        console.log(`     "${issue.preview}..."`);
      }
      if (issue.transcriptIndex !== undefined) {
        console.log(`   - Transcript[${issue.transcriptIndex}] (${issue.speaker}): Has thoughts field`);
        console.log(`     "${issue.thoughtsPreview}..."`);
      }
    }
  }

  console.log(`\nüí° To fix a soul, run: fixSoul('${soulIds[0]}')\n`);

  return soulIds;
}

function fixSoul(soulId) {
  if (typeof game === 'undefined' || !game.world) {
    console.error('‚ùå Game not loaded');
    return false;
  }

  const soul = game.world.getEntity(soulId);
  if (!soul) {
    console.error(`‚ùå Soul ${soulId} not found`);
    return false;
  }

  const event = soul.getComponent('soul_creation_event');
  if (!event) {
    console.error(`‚ùå Soul has no creation event`);
    return false;
  }

  const identity = soul.getComponent('soul_identity');
  const trueName = identity?.true_name || 'Unknown';

  console.log(`\nüîß Fixing soul: "${trueName}" (${soulId.substring(0, 8)}...)\n`);

  let fixed = 0;

  // Fix statements: remove <think> tags
  if (event.creationDebate?.statements) {
    for (let i = 0; i < event.creationDebate.statements.length; i++) {
      const stmt = event.creationDebate.statements[i];
      if (stmt.statement && stmt.statement.includes('<think>')) {
        const before = stmt.statement;
        stmt.statement = stmt.statement
          .replace(/<think>[\s\S]*?<\/think>/gi, '')
          .trim();
        console.log(`   ‚úì Removed <think> tag from statement[${i}] (${stmt.fate})`);
        fixed++;
      }
    }
  }

  // Fix transcript: remove thoughts field
  if (event.conversationTranscript) {
    for (let i = 0; i < event.conversationTranscript.length; i++) {
      const exchange = event.conversationTranscript[i];
      if (exchange.thoughts) {
        delete exchange.thoughts;
        console.log(`   ‚úì Removed thoughts from transcript[${i}] (${exchange.speaker})`);
        fixed++;
      }
    }
  }

  if (fixed === 0) {
    console.log('   ‚ÑπÔ∏è  Soul was already clean\n');
    return false;
  }

  // Mark as cleaned (Conservation of Game Matter)
  if (!soul.hasComponent('corruption_history')) {
    soul.addComponent({
      type: 'corruption_history',
      corruptions: []
    });
  }

  const history = soul.getComponent('corruption_history');
  history.corruptions.push({
    type: 'thinking_data_removed',
    cleaned_at: Date.now(),
    fixed_count: fixed,
    cleaner: 'manual_fix_script'
  });

  console.log(`\n‚úÖ Fixed ${fixed} corruption points in soul "${trueName}"\n`);
  console.log(`üíæ Remember to save the game to persist these changes!\n`);

  return true;
}

function fixAllSouls() {
  const corrupted = findCorruptedSouls();
  if (corrupted.length === 0) {
    return;
  }

  console.log(`\nüîß Fixing all ${corrupted.length} corrupted souls...\n`);

  let fixedCount = 0;
  for (const soulId of corrupted) {
    if (fixSoul(soulId)) {
      fixedCount++;
    }
  }

  console.log(`\n‚úÖ Fixed ${fixedCount} souls\n`);
  console.log(`üíæ Don't forget to SAVE THE GAME!\n`);
}
